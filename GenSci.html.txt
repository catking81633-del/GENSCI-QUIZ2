  // Photosynthesis

  q12:{ans:"B", expl:"Photosynthesis occurs in chloroplasts."},
  q13:{ans:"A", expl:"Chlorophyll is the main pigment absorbing light."},

  // Respiration
  q21:{ans:"B", expl:"ATP provides energy for cell work."},

  q25:{ans:"C", expl:"Respiration releases energy in ATP from glucose."},
  q26:{ans:"B", expl:"Anaerobic respiration in muscles produces lactic acid."},
  q27:{ans:"B", expl:"CO₂ is mainly produced in the Krebs cycle."},
  q28:{ans:"B", expl:"Oxygen is the final electron acceptor in ETC."},
  q29:{ans:"B", expl:"Glycolysis produces net 2 ATP."},
  q30:{ans:"B", expl:"Products are CO₂, H₂O, and ATP."}
};

function checkAnswers() {
  let score = 0;
  const form = document.getElementById("quizForm");
  const resultDiv = document.getElementById("result");
  resultDiv.innerHTML = "";

  for (let q in answerKey) {
    const selected = form.querySelector(`input[name="${q}"]:checked`);
    const correct = answerKey[q].ans;
    const explanation = answerKey[q].expl;
    const questionDiv = form.querySelector(`input[name="${q}"]`).closest(".question");

    let oldExpl = questionDiv.querySelector(".explanation");
    if (oldExpl) oldExpl.remove();

    const explPara = document.createElement("p");
    explPara.className = "explanation";

    if (selected) {
      if (selected.value === correct) {
        score++;
        questionDiv.style.border = "2px solid green";
        explPara.innerHTML = `<span class="correct">Correct!</span> ${explanation}`;
      } else {
        questionDiv.style.border = "2px solid red";
        explPara.innerHTML = `<span class="incorrect">Incorrect.</span> Correct: <b>${correct}</b>. ${explanation}`;
      }
    } else {
      questionDiv.style.border = "2px solid orange";
      explPara.innerHTML = `<span class="incorrect">No answer selected.</span> Correct: <b>${correct}</b>. ${explanation}`;
    }

    questionDiv.appendChild(explPara);
  }

  resultDiv.innerHTML = `<h2>Your Score: ${score} / ${Object.keys(answerKey).length}</h2>`;
}
</script>
